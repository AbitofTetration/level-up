Let's make a game!
	name:Level-Up
	by:Despacit
	desc:A game where you click for coins to level up.
	created:25/7/2017
	updated:24/10/2017
	version:1

Settings
	building cost increase:115%
	building cost refund:50%
	spritesheet:icons, 48 by 48, stuff/bunnyIcons.png
	stylesheet:stuff/bigBlue.css

Layout
	use default
		
Buttons
	*coinButton
		name:Click button
		desc:Click this little bunny to get more bunnies!
		on click:anim icon wobble
		on click:yield 1 coin
		on click:yield 1 xp
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red
		
Resources
	*coin|coins
		name:Coin|Coins
		desc:Buy coin upgrades to make coin production better.
		class:noBackground
		show earned
	*xp
		name:XP
		desc:Get more EXP to level up.
		class:noBackground
		show earned

	*magnet|magnets
		name:Magnet
		desc:Buy magnet upgrades to boost various facets of the game!
		class:noBackground
		show earned

Buildings
	*TEMPLATE
		on click:anim glow
		
	*coinBoost
		name:Coin Boost
		desc:Provides a boost to coin gain.
		cost increase:450%
		cost:10 coins
		passive:multiply yield of coins by 1.5
		unlocked
	*fasterLevel
		name:Faster Leveling
		desc:Level up faster!
		cost increase:800%
		cost:10 coins
		passive:lower cost of levelUp by 1
		unlocked
	*levelUp
		name:Level Up
		desc:Get Magnets!
		cost increase:150%
		cost:20 xp
		on earn:yield 1 magnet
		passive:increase levelUp yield of magnet by 1
		unlocked
